import java.util.Scanner;

public class OddEven {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);  s
        System.out.print("Enter a number: ");      
        int num = scanner.nextInt();               

      
        if (num % 2 == 0) {

        }

        scanner.close();
